                public function getAssignedhists($status=1)
{
    return $this->hasMany(HistoricalFact::className(), ['id' => 'histId'])->viaTable('historicalAssign', ['userId' => 'id'], 
        function($query) {
            $query->onCondition(['type' => 2]);
        }
    )->andOnCondition(['status' => $status]);
}
